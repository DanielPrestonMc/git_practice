SELECT c.continent, py.year, SUM(py.population)
FROM countries c
JOIN population_years py
	ON c.id = py.country_id
WHERE year = 2005
	AND c.continent = "Oceania";

SELECT c.continent, py.year, AVG(py.population)
FROM countries c
JOIN population_years py
	ON c.id = py.country_id
WHERE year = 2003
	AND c.continent = "South America";
